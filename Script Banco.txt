use Pizzaria;

create table Pizzas(
	id int,
	nome varchar(50),
	ingredientes varchar(500),
	preco float,
    primary key (id)
);

create table Clientes(
	cpf varchar(11),
    nome varchar(100),
	logradouro varchar(100),
	bairro varchar(100),
	cidade varchar(100),
    telefone varchar(20),
    primary key (cpf)
);

create table Pedidos (
	id int primary key,
    cpfCliente varchar(11),
    idPizza int,
	quantidade int,
    status varchar(15),
	foreign key (idPizza) references Pizzas(id),
    foreign key (cpfCliente) references Clientes(cpf)
);
